double f1(double x)
{
	double y = 0;
	y = 2 * pow(x, 3) + 5;
	return y;
}

double f2(double x)
{
	double y = 0;
	y = pow((x - 5), 2) + 4 * x + 4;
	return y;
}

double f3(double x)
{
	double y = 0;
	y = (5 / x) + 1;
	return y;
}


double integ(double(*V)(double x), double a, double b, double eps)
{
	double N, x, h, s3, s2, s1, s0 = 0, k = 0;
	s3 = eps + 1;
	for (N = 1; s3 > eps; N++)
	{
		s1 = 0;
		s2 = 0;
		s3 = 0;
		h = (b - a) / N;
		for (int i = 0; i < N; i++)
		{
			x = a + i * h;
			s0 = h * (*V)(x);
			s1 = s1 + s0;
		}

		h = (b - a) / (N * 2);
		for (int i = 0; i < (N * 2); i++)
		{
			x = a + i * h;
			s0 = h * (*V)(x);
			s2 = s2 + s0;
		}

		s3 = (s2 - s1);
	}

	return s2;
}

int main()
{
	setlocale(LC_CTYPE, "");
	double a, b, eps, s, temp = 0;
	int n;
	puts("Выберите функцию:\n 1) y=4*x^3+8 \n 2) y=(x-3)^2+5x+2\n 3) y=3/x+2\n\n Функция N ");
	do
	{
		scanf("%d", &n);
	} while ((n < 1) || (n > 3));
	printf("\n");
	puts("Введите a");
	scanf("%lf", &a);
	puts("Введите b");
	do
		scanf("%lf", &b);
	while (b == a);
	puts("Введите точность");
	scanf("%lf", &eps);
	printf("\n");

	if (b < a)
	{
		temp = a;
		a = b;
		b = temp;
	}

	if (n == 1)
		s = integ(f1, a, b, eps);
	else
	{
		if (n == 2)
			s = integ(f2, a, b, eps);
		else
			s = integ(f3, a, b, eps);
	}
	printf("Значение интеграла= %.4lf\n", s);

	getch();
	return 0;
}
