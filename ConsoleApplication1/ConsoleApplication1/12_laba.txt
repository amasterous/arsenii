void output1(char *s)
{
	for (int i = 0; i < strlen(s); i++)
		printf("\n %c", s[i]);
}

void word_output(char *p)
{
	int i = 0;
	while (p[i] != '\0')
	{
		if (p[i] == ' ')
			printf("\n");
		else
			printf("%c", p[i]);
		i++;
	}
}
